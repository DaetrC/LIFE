function doGet(e) {
  const action = (e.parameter && e.parameter.action) || 'ping';
  if (action === 'readAll') return respond(readAll());
  if (action === 'getCalendar') return respond(getCalendarEvents(e.parameter));
  return respond({ status:'ok', message:'Life OS backend running' });
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    if (body.action === 'writeAll') return respond(writeAll(body.data));
    if (body.action === 'addEvent') return respond(addCalEvent(body));
    return respond({ status:'error', message:'Unknown action' });
  } catch(err) {
    return respond({ status:'error', message:err.toString() });
  }
}

function readAll() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const tab = ss.getSheetByName('os_data');
  if (!tab) return { status:'error', message:'Run setupSheets() first' };
  const raw = tab.getRange('A2').getValue();
  return { status:'ok', data: raw ? JSON.parse(raw) : {} };
}

function writeAll(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let tab = ss.getSheetByName('os_data');
  if (!tab) tab = ss.insertSheet('os_data');
  tab.getRange('A1').setValue('data_json');
  tab.getRange('B1').setValue('last_updated');
  tab.getRange('A2').setValue(JSON.stringify(data));
  tab.getRange('B2').setValue(new Date().toISOString());
  return { status:'ok', saved:true };
}

function getCalendarEvents(params) {
  try {
    const days = parseInt(params.days) || 14;
    const now = new Date();
    const end = new Date(now.getTime() + days * 86400000);
    const cals = CalendarApp.getAllCalendars();
    const events = [];
    cals.forEach(cal => {
      cal.getEvents(now, end).forEach(ev => {
        events.push({
          id: ev.getId(),
          title: ev.getTitle(),
          start: ev.getStartTime().toISOString(),
          end: ev.getEndTime().toISOString(),
          allDay: ev.isAllDayEvent(),
          calendar: cal.getName(),
          color: cal.getColor(),
          location: ev.getLocation() || '',
          desc: ev.getDescription() || ''
        });
      });
    });
    events.sort((a,b) => new Date(a.start) - new Date(b.start));
    return { status:'ok', events };
  } catch(err) {
    return { status:'error', message:err.toString() };
  }
}

function addCalEvent(body) {
  try {
    const cal = CalendarApp.getDefaultCalendar();
    const start = new Date(body.start);
    const end = new Date(body.end || body.start);
    const ev = cal.createEvent(body.title, start, end, { description: body.desc || '', location: body.location || '' });
    return { status:'ok', id: ev.getId() };
  } catch(err) {
    return { status:'error', message:err.toString() };
  }
}

function respond(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function setupSheets() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let tab = ss.getSheetByName('os_data');
  if (!tab) tab = ss.insertSheet('os_data');
  tab.getRange('A1:B1').setValues([['data_json','last_updated']]);
  const r = tab.getRange('A1:B1');
  r.setBackground('#0A0A0A'); r.setFontColor('#FFFFFF'); r.setFontWeight('bold');
  SpreadsheetApp.getUi().alert('Setup complete! Now deploy as Web App.');
}

}
